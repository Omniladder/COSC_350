       	int parent_Iterations = string_To_Int(Inputs[]);

	pid = fork();
	switch(pid)
	{
	case -:
		perror("fork failed");
		exit();
	case :
		message = "This is the child";
		n = child_Iterations;
		break;
	default:
		message = "This is the parent";
		n = parent_Iterations;
		break;
	}

	for(; n > ; n--)
	{
		if(pid == )
		{	
			puts(message);
			sleep(child_Sleep);
		}
		else
		{
			puts(message);
			sleep(parent_Sleep);
		}
	}
	exit();
}
